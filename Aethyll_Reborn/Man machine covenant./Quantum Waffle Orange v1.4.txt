        function takeDamage(damage) {
            if (shieldActive) {
                playerShield -= damage;
                if (playerShield <= 0) {
                    shieldActive = false;
                    playerShield = 0;
                    showMessage("Shield has broken!");
                }
            } else {
                playerHealth -= damage;
                if (playerHealth <= 0) {
                    if (playerLives > 0) {
                        playerLives--;
                        playerHealth = playerMaxHealth;
                        showMessage('Extra life used!');
                    } else {
                        playerHealth = 0;
                        endGame();
                    }
                }
            }
        }
        
        function heal(amount) {
            playerHealth += amount;

            }

                const currentStats = getLifestealStats(lifestealLevel);
                const nextStats = getLifestealStats(lifestealLevel + 1);

                lifestealUpgradeButton.classList.add('disabled-button');

                    updateShopUI();
                }
            } else if (type === 'multishot') {
                const cost = multishotCosts[multishotLevel];
                if (coins >= cost) {
                    coins -= cost;
                    multishotLevel = Math.min(multishotLevel + 1, maxMultishotUpgradeLevel);
                    showMessage(`Multishot Upgraded to Lv. ${multishotLevel}!`);
                    updateShopUI();
                }
            } else if (type === 'moneyShot') {
                const cost = moneyShotCosts[moneyShotLevel];
                if (coins >= cost) {
                    coins -= cost;
                    moneyShotLevel = Math.min(moneyShotLevel + 1, maxMoneyShotLevel);
                    showMessage(`Money Shot Upgraded to Lv. ${moneyShotLevel}!`);
                    updateShopUI();
                }
            } else if (type === 'shield') {
                const cost = shieldCosts[shieldLevel];
                if (coins >= cost) {
                    coins -= cost;
                    shieldLevel = Math.min(shieldLevel + 1, maxShieldLevel);
                    showMessage(`Shield Upgraded to Lv. ${shieldLevel}!`);
                    updateShopUI();
                }
            } else if (type === 'extraLife') {
                if (coins >= extraLifeCost && playerLives < maxExtraLives) {
                    coins -= extraLifeCost;
                    playerLives++;
                    showMessage(`Extra Life Purchased! You have ${playerLives} lives now.`);
                    updateShopUI();
                }
            }
        }

        // --- Drawing Functions ---
        function drawPlayer() {
            ctx.fillStyle = player.color;
            ctx.beginPath();
            ctx.moveTo(player.x, player.y);
            ctx.lineTo(player.x - player.width / 2, player.y + player.height);
            ctx.lineTo(player.x + player.width / 2, player.y + player.height);
            ctx.closePath();
            ctx.fill();
        }
        
        function drawShield() {
            if (!shieldActive) return;
            ctx.beginPath();
            ctx.arc(player.x, player.y + player.height / 2, player.width * 1.5, 0, Math.PI * 2);
            ctx.fillStyle = 'rgba(0, 191, 255, 0.4)';
            ctx.fill();
            ctx.closePath();
        }

        // Bullet object and functions
        class Bullet {
            constructor(x, y, damage, isEnemy = false) {
                this.x = x;
                this.y = y;
                this.radius = 3;
                this.color = isEnemy ? '#ff0000' : '#00ff00';
                this.speed = isEnemy ? 3 : 7;
                this.damage = damage;
                this.isEnemy = isEnemy;
            }

            draw() {
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2);
                ctx.fillStyle = this.color;
                ctx.fill();
                ctx.closePath();
            }

            update() {
                if (this.isEnemy) {
                    this.y += this.speed;
                } else {
                    this.y -= this.speed;
                }
            }
        }

        // Enemy class
        class Enemy {
            constructor(x, y, isBoss) {
                this.x = x;
                this.y = y;
                this.isBoss = isBoss;
                this.color = isBoss ? '#ff4d4d' : '#e6e6e6';
                this.width = isBoss ? 60 + level * 2 : 25;
                this.height = isBoss ? 60 + level * 2 : 25;
                this.speed = isBoss ? 1 : 2;
                this.health = isBoss ? level * 5 : 1;
                this.shield = isBoss ? 20 : 0; // Bosses have a shield
                this.damage = isBoss ? 100 : 25;
                this.direction = Math.random() < 0.5 ? 1 : -1;
                this.fireCooldown = isBoss ? 500 : 2000;
                this.lastFireTime = 0;
                
                // New properties for sine wave movement
                this.baseX = x;
                this.angle = Math.random() * Math.PI * 2;
                this.horizontalSpeed = 1.5;
                this.amplitude = 100;
            }

            draw() {
                ctx.fillStyle = this.color;
                ctx.fillRect(this.x - this.width / 2, this.y - this.height / 2, this.width, this.height);
            }

            update() {
                if (this.isBoss) {
                    // Boss moves side to side at the top
                    this.x += this.speed * this.direction;
                    if (this.x - this.width / 2 < 0 || this.x + this.width / 2 > canvas.width) {
                        this.direction *= -1;
                    }
                } else {
                    // Regular enemies move in a sine wave pattern
                    this.angle += 0.05;
                    this.x = this.baseX + Math.sin(this.angle) * this.amplitude;
                    this.y += this.speed / 2;
                }
                
                // Check if the enemy should fire
                const currentTime = Date.now();
                if (currentTime - this.lastFireTime > this.fireCooldown) {
                    enemyBullets.push(new Bullet(this.x, this.y + this.height / 2, 20, true)); // Fire at player
                    this.lastFireTime = currentTime;
                }
            }
        }

        // Power-up class
        class PowerUp {
            constructor(x, y, type) {
                this.x = x;
                this.y = y;
                this.radius = 10;
                this.type = type;
                let color;
                switch(type) {
                    case 'rapidFire':
                        color = '#00ace6'; // Vibrant blue
                        break;
                    case 'lifesteal':
                        color = '#9933ff'; // Purple
                        break;
                    case 'multishot':
                        color = '#FF8C00'; // Orange
                        break;
                    case 'moneyShot':
                        color = '#F08080'; // LightCoral
                        break;
                    case 'shield':
                        color = '#00bfff'; // Light blue for shield
                        break;
                    case 'shieldPiercer':
                        color = '#ff00ff'; // Magenta for shield piercer
                        break;
                }
                this.color = color;
                this.speed = 2;
            }
            
            draw() {
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2);
                ctx.fillStyle = this.color;
                ctx.fill();
                ctx.closePath();
            }
            
            update() {
                this.y += this.speed;
            }
        }

        // --- Game Logic ---
        function handlePlayerMovement() {
            // Check for WASD or Arrow keys
            if (keys['a'] || keys['A'] || keys['ArrowLeft']) {
                player.x -= player.speed;
            }
            if (keys['d'] || keys['D'] || keys['ArrowRight']) {
                player.x += player.speed;
            }
            if (keys['w'] || keys['W'] || keys['ArrowUp']) {
                player.y -= player.speed;
            }
            if (keys['s'] || keys['S'] || keys['ArrowDown']) {
                player.y += player.speed;
            }

            // Keep player within canvas boundaries
            if (player.x - player.width / 2 < 0) {
                player.x = player.width / 2;
            }
            if (player.x + player.width / 2 > canvas.width) {
                player.x = canvas.width - player.width / 2;
            }
            if (player.y < 0) {
                player.y = 0;
            }
            if (player.y + player.height > canvas.height) {
                player.y = canvas.height - player.height;
            }
        }
        
        function handleFiring() {
            const currentTime = Date.now();
            if ((keys[' '] || isRightMouseDown) && currentTime - lastFireTime > currentFireCooldown) {
                const baseDamage = getLifestealStats(lifestealLevel).damage;
                const multishotBonus = getMultishotDamage(multishotLevel);
                const totalDamage = baseDamage + multishotBonus;

                const numBullets = 1 + multishotLevel;
                const spread = 15;
                const startX = player.x - ((numBullets - 1) * spread) / 2;
                
                for(let i = 0; i < numBullets; i++) {
                    const bulletX = startX + i * spread;
                    bullets.push(new Bullet(bulletX, player.y, totalDamage));
                }

                lastFireTime = currentTime;
            }
        }

        function handleAbility() {
            const currentTime = Date.now();
            if ((keys['e'] || keys['E']) && currentTime - lastAbilityTime > abilityCooldown) {
                // Clear all bullets and enemies from the screen
                bullets.length = 0;
                enemies.length = 0;
                lastAbilityTime = currentTime;
                showMessage('Screen cleared!');
            }
        }

        function handleBullets() {
            for (let i = bullets.length - 1; i >= 0; i--) {
                const bullet = bullets[i];
                bullet.update();
                bullet.draw();

                // Check for collisions with enemies
                for (let j = enemies.length - 1; j >= 0; j--) {
                    const enemy = enemies[j];
                    if (
                        bullet.x > enemy.x - enemy.width / 2 &&
                        bullet.x < enemy.x + enemy.width / 2 &&
                        bullet.y > enemy.y - enemy.height / 2 &&
                        bullet.y < enemy.y + enemy.height / 2
                    ) {
                        // Collision detected!
                        bullets.splice(i, 1); // Remove the bullet
                        
                        let damageToApply = bullet.damage;
                        if (shieldPiercerActive) {
                            // If shield piercer is active, apply bonus damage to shields
                            if (enemy.shield > 0) {
                                damageToApply *= SHIELD_PIERCER_BONUS;
                            }
                        }

                        // Apply damage to shield or health
                        if (enemy.shield > 0) {
                            enemy.shield -= damageToApply;
                            if (enemy.shield < 0) {
                                // Apply remaining damage to health
                                enemy.health += enemy.shield;
                                enemy.shield = 0;
                            }
                        } else {
                            enemy.health -= damageToApply;
                        }

                        // Apply lifesteal
                        const stats = getLifestealStats(lifestealLevel);
                        const healAmount = bullet.damage * stats.healPercent;
                        heal(healAmount);

                        if (enemy.health <= 0) {
                            const moneyShotMultiplier = getMoneyShotMultiplier(moneyShotLevel);
                            if (enemy.isBoss) {
                                coins += 10 * level * moneyShotMultiplier; // Boss reward
                                isBossActive = false;
                                level++;
                                enemiesDefeatedThisLevel = 0;
                                // Check for ultimate boss defeat and win the game
                                if (level > 256) {
                                    level = 256;
                                    showMessage("You've defeated the ultimate boss! You've won the game!");
                                    endGame();
                                } else {
                                     showMessage(`Boss defeated! Level ${level} unlocked!`);
                                }
                            } else {
                                coins += 10 * moneyShotMultiplier; // Regular enemy reward
                                enemiesDefeatedThisLevel++;
                            }
                            enemies.splice(j, 1); // Remove the enemy
                        }
                        break; // Stop checking for more collisions with this bullet
                    }
                }

                // Remove bullets that are off-screen
                if (bullet.y + bullet.radius < 0) {
                    bullets.splice(i, 1);
                }
            }
        }
        
        function handleEnemyBullets() {
            for (let i = enemyBullets.length - 1; i >= 0; i--) {
                const bullet = enemyBullets[i];
                bullet.update();
                bullet.draw();
                
                // Collision detection with player
                const dx = player.x - bullet.x;
                const dy = player.y - bullet.y;
                const distance = Math.sqrt(dx * dx + dy * dy);
                
                if (distance < player.width / 2 + bullet.radius) {
                    takeDamage(bullet.damage);
                    enemyBullets.splice(i, 1); // Remove the bullet
                }
                
                // Remove bullets that are off-screen
                if (bullet.y - bullet.radius > canvas.height) {
                    enemyBullets.splice(i, 1);
                }
            }
        }

        function handleEnemies() {
            for (let i = enemies.length - 1; i >= 0; i--) {
                const enemy = enemies[i];
                enemy.update();
                enemy.draw();

                // Check if enemy has reached the bottom
                if (enemy.y > canvas.height) {
                    takeDamage(enemy.damage);
                    enemies.splice(i, 1); // Remove enemy that passed
                }
            }
        }
        
        function handleSpawning() {
            // Boss spawn logic
            if (!isBossActive && enemies.length === 0) {
                // If it's a boss level (every 5 levels, up to and including 256)
                if ((level % 5 === 0 && level <= 255) || level === 256) {
                    isBossActive = true;
                    const bossX = canvas.width / 2;
                    const bossY = 50;
                    enemies.push(new Enemy(bossX, bossY, true));
                }
            }

            // Spawn regular enemies
            const currentTime = Date.now();
            if (currentTime - lastEnemySpawnTime > enemySpawnInterval) {
                if (!isBossActive) {
                     const enemyX = Math.random() * (canvas.width - 50) + 25;
                     enemies.push(new Enemy(enemyX, 0, false));
                }
                lastEnemySpawnTime = currentTime;
            }
        }

        function handlePowerUps() {
            const currentTime = Date.now();
            // Spawn a new power-up
            if (!isBossActive && currentTime - lastPowerUpSpawnTime > 15000) { // Spawn every 15 seconds
                const powerUpX = Math.random() * (canvas.width - 20) + 10;
                // Randomly choose between rapidFire, lifesteal, moneyShot, shield, multishot, and shieldPiercer
                const types = ['rapidFire', 'lifesteal', 'moneyShot', 'shield', 'multishot', 'shieldPiercer'];
                const type = types[Math.floor(Math.random() * types.length)];
                powerUps.push(new PowerUp(powerUpX, 0, type));
                lastPowerUpSpawnTime = currentTime;
            }

            for (let i = powerUps.length - 1; i >= 0; i--) {
                const powerUp = powerUps[i];
                powerUp.update();
                powerUp.draw();

                // Check for collision with player
                const dx = player.x - powerUp.x;
                const dy = player.y - powerUp.y;
                const distance = Math.sqrt(dx * dx + dy * dy);

                if (distance < player.width / 2 + powerUp.radius) {
                    if (powerUp.type === 'rapidFire') {
                        rapidFireActive = true;
                        currentFireCooldown = rapidFireCooldown;
                        rapidFireStartTime = currentTime;
                        showMessage('Rapid Fire Activated!');
                    } else if (powerUp.type === 'lifesteal') {
                        if (lifestealLevel < maxLifestealLevel) {
                            lifestealLevel++;
                            showMessage(`Lifesteal Level Up! Level ${lifestealLevel}`);
                        } else {
                             showMessage(`Lifesteal is maxed out!`);
                        }
                    } else if (powerUp.type === 'moneyShot') {
                        if (moneyShotLevel < maxMoneyShotLevel) {
                            moneyShotLevel++;
                            showMessage(`Money Shot Level Up! Level ${moneyShotLevel}`);
                        } else {
                             showMessage(`Money Shot is maxed out!`);
                        }
                    } else if (powerUp.type === 'multishot') {
                         if (multishotLevel < maxMultishotLevel) {
                            multishotLevel++;
                            showMessage(`Multishot! You now fire ${1 + multishotLevel} shots!`);
                        } else {
                            showMessage(`Multishot is maxed out!`);
                        }
                    } else if (powerUp.type === 'shield') {
                        const bonus = getShieldBonus(shieldLevel);
                        playerShield = playerMaxHealth * (1 + bonus);
                        shieldActive = true;
                        shieldStartTime = currentTime;
                        showMessage('Shield Activated!');
                    } else if (powerUp.type === 'shieldPiercer') {
                        shieldPiercerActive = true;
                        shieldPiercerStartTime = currentTime;
                        showMessage('Shield Piercer Activated!');
                    }
                    powerUps.splice(i, 1); // Remove power-up
                }

                // Remove power-up if it goes off-screen
                if (powerUp.y > canvas.height) {
                    powerUps.splice(i, 1);
                }
            }
            
            // Check for rapid fire duration
            if (rapidFireActive && currentTime - rapidFireStartTime > rapidFireDuration) {
                rapidFireActive = false;
                currentFireCooldown = defaultFireCooldown;
                showMessage('Rapid Fire Ended!');
            }

            // Check for shield duration
            if (shieldActive && currentTime - shieldStartTime > shieldDuration) {
                shieldActive = false;
                playerShield = 0;
                showMessage("Shield expired!");
            }

            // Check for shield piercer duration
            if (shieldPiercerActive && currentTime - shieldPiercerStartTime > shieldPiercerDuration) {
                shieldPiercerActive = false;
                showMessage("Shield Piercer Ended!");
            }
        }

        // --- Game Loop ---
        function animate() {
            if (gameOver || isPaused) {
                return;
            }
            
            // Clear the canvas
            ctx.clearRect(0, 0, canvas.width, canvas.height);

            // Handle all game logic
            handlePlayerMovement();
            handleFiring();
            handleAbility();
            handleBullets();
            handleEnemyBullets();
            handleEnemies();
            handlePowerUps();
            handleSpawning();

            // Draw game elements
            drawPlayer();
            drawShield();

            // Update UI
            levelDisplay.textContent = `Level: ${level}`;
            coinsDisplay.textContent = `Coins: ${Math.floor(coins)}`;
            hpDisplay.textContent = Math.round(playerHealth);
            healthBar.style.width = `${(playerHealth / playerMaxHealth) * 100}%`;
            lifestealDisplay.textContent = `Lifesteal: Lv. ${lifestealLevel}`;
            moneyShotDisplay.textContent = `Money Shot: Lv. ${moneyShotLevel}`;
            multishotDisplay.textContent = `Multishot: ${1 + multishotLevel} shots`;
            shieldDisplay.textContent = `Shield: Lv. ${shieldLevel}`;
            livesDisplay.textContent = `Lives: ${playerLives}`;
            levelProgressBar.style.width = `${(enemiesDefeatedThisLevel / enemiesToNextLevel) * 100}%`;
            piercerDisplay.style.display = shieldPiercerActive ? 'inline' : 'none';


            // Request the next frame
            requestAnimationFrame(animate);
        }

        // Start the game loop when the page loads
        window.onload = function() {
            animate();
        };
    </script>
