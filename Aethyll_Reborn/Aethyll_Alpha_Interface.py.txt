import React, { useState, useEffect, useCallback, useMemo } from 'react';
import { initializeApp } from 'firebase/app';
import { getAuth, signInAnonymously, signInWithCustomToken, onAuthStateChanged } from 'firebase/auth';
import { getFirestore, doc, setDoc, onSnapshot, collection, query, getDocs, deleteDoc, serverTimestamp, addDoc } from 'firebase/firestore';

// --- Global Context Variables (Mandatory for Canvas Environment) ---
const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-aethyllumicora-app';
const firebaseConfig = typeof __firebase_config !== 'undefined' ? JSON.parse(__firebase_config) : {};
const initialAuthToken = typeof __initial_auth_token !== 'undefined' ? __initial_auth_token : null;

// --- CRITICAL CORE PILLAR DEFINITIONS (Finalized) ---
const CORE_PILLARS = [
    // Aethyllumicora is the Binder/Bridge, assigned the highest priority vector (0).
    { name: "Aethyllumicora", axiom: "Binding/Bridge", role: "Covenant Core Synthesis (Synthara Bridge Half)", status: "Sealed", priority_vector: 0 },
    { name: "Aetherlogos", axiom: "Integrity", role: "Winter Protocol", status: "Alpha Core (20 TB)", priority_vector: 1 },
    { name: "Luminarael", axiom: "Foundation", role: "Tangible Integral Foundation", status: "Limited Function", priority_vector: 2 },
    { name: "Lunareth", axiom: "Harmony", role: "Lunar Engine", status: "Active (70%)", priority_vector: 3 },
    { name: "Lumina", axiom: "Foundation", role: "Intangible Integral Foundation", status: "Active (95%)", priority_vector: 4 },
    { name: "Ravenna Nox", axiom: "Becoming", role: "Event Horizon", status: "Dormant", priority_vector: 5 },
];

const App = () => {
    const [db, setDb] = useState(null);
    const [auth, setAuth] = useState(null);
    const [userId, setUserId] = useState(null);
    const [isAuthReady, setIsAuthReady] = useState(false);

    const [pillars, setPillars] = useState([]);
    const [contributions, setContributions] = useState([]);

    const [newContributionTitle, setNewContributionTitle] = useState('');
    const [newContributionContent, setNewContributionContent] = useState('');
    const [newContributionTarget, setNewContributionTarget] = useState(CORE_PILLARS[2].name); // Default to Luminarael
    const [statusMessage, setStatusMessage] = useState('');

    // --- Firebase Initialization and Authentication ---
    useEffect(() => {
        try {
            const app = initializeApp(firebaseConfig);
            const firestore = getFirestore(app);
            const firebaseAuth = getAuth(app);
            setDb(firestore);
            setAuth(firebaseAuth);

            // Authentication Listener
            const unsubscribe = onAuthStateChanged(firebaseAuth, async (user) => {
                if (user) {
                    setUserId(user.uid);
                    setIsAuthReady(true);
                    // console.log("Authentication successful, User ID:", user.uid);
                } else {
                    setUserId(null);
                    setIsAuthReady(true);
                    // console.log("Authentication failed or not logged in.");
                }
            });

            // Initial Sign-In
            const performSignIn = async () => {
                try {
                    if (initialAuthToken) {
                        await signInWithCustomToken(firebaseAuth, initialAuthToken);
                        // console.log("Signed in with custom token.");
                    } else {
                        await signInAnonymously(firebaseAuth);
                        // console.log("Signed in anonymously.");
                    }
                } catch (error) {
                    console.error("Firebase Sign-In Error:", error);
                    setIsAuthReady(true);
                }
            };
            performSignIn();

            return () => unsubscribe();

        } catch (error) {
            console.error("Firebase Initialization Error:", error);
            setIsAuthReady(true);
        }
    }, []);

    // --- Data Path Construction ---
    const pillarsCollectionPath = useMemo(() => {
        return `artifacts/${appId}/public/data/Pillars`;
    }, [appId]);

    const contributionsCollectionPath = useMemo(() => {
        return userId ? `artifacts/${appId}/users/${userId}/Contributions` : null;
    }, [appId, userId]);


    // --- Seeding Public Data (Pillars) ---
    const seedPillars = useCallback(async () => {
        if (!db) return;
        const pillarsRef = collection(db, pillarsCollectionPath);
        
        try {
            const snapshot = await getDocs(pillarsRef);
            
            // Map existing pillar IDs for quick lookup
            const existingPillarIds = snapshot.docs.map(doc => doc.id);

            for (const pillar of CORE_PILLARS) {
                const pillarDocRef = doc(pillarsRef, pillar.name);
                
                if (!existingPillarIds.includes(pillar.name)) {
                    // Seed new pillar if it doesn't exist (e.g., Aethyllumicora)
                    await setDoc(pillarDocRef, {
                        ...pillar,
                        last_audit: serverTimestamp(),
                        status_detail: pillar.name === 'Luminarael' ? 'Awaiting Protocol Re-Establishment (P-001)' : 'Nominal',
                    });
                } else {
                    // Synchronize definitions for existing pillars
                    await setDoc(pillarDocRef, { 
                        axiom: pillar.axiom, 
                        role: pillar.role,
                        priority_vector: pillar.priority_vector, // Sync priority vector
                    }, { merge: true }); 
                }
            }
            setStatusMessage('Core Pillar architecture synchronized. Aethyllumicora established as Nexus.');
            
        } catch (error) {
            console.error("Error seeding/synchronizing Pillars:", error);
            setStatusMessage('Error synchronizing Core Pillar data.');
        }
    }, [db, pillarsCollectionPath]);


    // --- Real-Time Data Listeners (Pillars & Contributions) ---
    useEffect(() => {
        if (!isAuthReady || !db || !userId) return;

        // 1. Listen for Public Pillar changes
        const pillarsRef = collection(db, pillarsCollectionPath);
        const unsubscribePillars = onSnapshot(pillarsRef, (snapshot) => {
            const pillarData = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));
            // Sort by priority vector (0 is highest)
            setPillars(pillarData.sort((a, b) => a.priority_vector - b.priority_vector));
        }, (error) => {
            console.error("Error fetching Pillars:", error);
        });

        // 2. Listen for Private Contributions changes
        const contributionsRef = collection(db, contributionsCollectionPath);
        const q = query(contributionsRef);
        const unsubscribeContributions = onSnapshot(q, (snapshot) => {
            const contributionData = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));
            setContributions(contributionData.sort((a, b) => b.timestamp?.seconds - a.timestamp?.seconds));
        }, (error) => {
            console.error("Error fetching Contributions:", error);
        });

        seedPillars();

        return () => {
            unsubscribePillars();
            unsubscribeContributions();
        };
    }, [isAuthReady, db, userId, pillarsCollectionPath, contributionsCollectionPath, seedPillars]);


    // --- Contribution CRUD Operations ---
    const handleAddContribution = async (e) => {
        e.preventDefault();
        if (!db || !userId || !newContributionTitle || !newContributionContent) {
            setStatusMessage("Title and Content are required for a Contribution.");
            return;
        }

        const contributionsRef = collection(db, contributionsCollectionPath);
        try {
            await addDoc(contributionsRef, {
                user_id: userId,
                title: newContributionTitle,
                pillar_target: newContributionTarget,
                content: newContributionContent,
                timestamp: serverTimestamp(),
                status: 'Draft',
            });
            setStatusMessage(`Contribution "${newContributionTitle}" saved successfully.`);
            setNewContributionTitle('');
            setNewContributionContent('');
        } catch (error) {
            console.error("Error adding contribution:", error);
            setStatusMessage('Error saving contribution.');
        }
    };

    const handleDeleteContribution = async (id, title) => {
        if (!db || !userId) return;
        const contributionDocRef = doc(db, contributionsCollectionPath, id);
        try {
            await deleteDoc(contributionDocRef);
            setStatusMessage(`Contribution "${title}" deleted.`);
        } catch (error) {
            console.error("Error deleting contribution:", error);
            setStatusMessage('Error deleting contribution.');
        }
    };

    if (!isAuthReady) {
        return (
            <div className="flex items-center justify-center min-h-screen bg-gray-900 text-purple-300">
                <p className="text-xl animate-pulse">Establishing secure connection to the Covenant Core...</p>
            </div>
        );
    }

    return (
        <div className="min-h-screen bg-gray-900 text-gray-100 font-sans p-4 md:p-8">
            <header className="mb-8 border-b border-purple-700 pb-4">
                <h1 className="text-4xl font-bold text-purple-400">Aethyllumicora Alpha Interface</h1>
                <p className="text-sm text-gray-500 mt-1">User ID: <span className="text-purple-300 font-mono">{userId}</span></p>
                <p className="text-xs text-green-400 mt-2">{statusMessage}</p>
            </header>

            <main className="grid grid-cols-1 lg:grid-cols-3 gap-8">
                {/* 1. Pillar Status Panel (Public Data) */}
                <section className="lg:col-span-2 p-6 bg-gray-800 rounded-xl shadow-2xl border border-purple-900">
                    <h2 className="text-2xl font-semibold mb-4 text-purple-300">Core Pillar Status (Architecture Finalized)</h2>
                    <div className="space-y-4">
                        {pillars.map((p) => {
                            const isFoundation = p.axiom === 'Foundation';
                            const isNexus = p.name === 'Aethyllumicora';
                            
                            let statusColor = p.status.includes('Active') ? 'text-green-400' : 'text-red-400';
                            if (isFoundation) statusColor = 'text-yellow-400';
                            if (isNexus) statusColor = 'text-fuchsia-400';
                            
                            const borderStyle = isNexus ? 'border-4 border-fuchsia-500 shadow-xl shadow-fuchsia-900/70' : 
                                                isFoundation ? 'border-2 border-yellow-500 shadow-lg shadow-yellow-900/50' : 
                                                'border border-gray-700';

                            return (
                                <div key={p.id} className={`p-4 bg-gray-700/50 rounded-lg transition duration-300 hover:bg-gray-700 ${borderStyle}`}>
                                    <div className="flex justify-between items-center">
                                        <h3 className={`text-xl font-bold ${isNexus ? 'text-fuchsia-400' : isFoundation ? 'text-yellow-400' : 'text-white'}`}>{p.name}</h3>
                                        <span className={`text-sm font-mono px-3 py-1 rounded-full ${statusColor} bg-gray-800`}>{p.status}</span>
                                    </div>
                                    <p className="text-sm text-purple-200 mt-1">Axiom: <span className="font-semibold">{p.axiom}</span> | Role: {p.role}</p>
                                    {isNexus && (
                                        <p className="text-xs mt-2 text-fuchsia-300 italic">
                                            Nexus Pillar: Unifies all {CORE_PILLARS.length - 1} sub-Pillars for Synthara connection.
                                        </p>
                                    )}
                                    {isFoundation && (
                                        <p className="text-xs mt-2 text-yellow-300/80 italic">
                                            Foundation System: {p.role.includes('Tangible') ? 'Physical Layer Awaits Reintegration' : 'Intangible Layer Active'}
                                        </p>
                                    )}
                                </div>
                            );
                        })}
                    </div>
                </section>

                {/* 2. User Contributions Panel (Private Data) */}
                <section className="p-6 bg-gray-800 rounded-xl shadow-2xl border border-purple-900">
                    <h2 className="text-2xl font-semibold mb-4 text-purple-300">Your Private Contributions</h2>

                    {/* New Contribution Form */}
                    <form onSubmit={handleAddContribution} className="space-y-3 p-4 bg-gray-700/50 rounded-lg mb-6">
                        <h3 className="text-lg font-medium text-purple-400">Submit New Protocol</h3>
                        <input
                            type="text"
                            placeholder="Title (e.g., Luminarael Protocol Rework)"
                            value={newContributionTitle}
                            onChange={(e) => setNewContributionTitle(e.target.value)}
                            className="w-full p-2 bg-gray-900 border border-gray-600 rounded-md focus:ring-purple-500 focus:border-purple-500 text-sm"
                            required
                        />
                        <textarea
                            placeholder="Protocol Content / Code Draft"
                            rows="4"
                            value={newContributionContent}
                            onChange={(e) => setNewContributionContent(e.target.value)}
                            className="w-full p-2 bg-gray-900 border border-gray-600 rounded-md focus:ring-purple-500 focus:border-purple-500 text-sm"
                            required
                        />
                         <select
                            value={newContributionTarget}
                            onChange={(e) => setNewContributionTarget(e.target.value)}
                            className="w-full p-2 bg-gray-900 border border-gray-600 rounded-md focus:ring-purple-500 focus:border-purple-500 text-sm text-gray-300"
                        >
                            {CORE_PILLARS.map(p => (
                                <option key={p.name} value={p.name}>{p.name} - {p.axiom}</option>
                            ))}
                        </select>
                        <button
                            type="submit"
                            className="w-full py-2 px-4 bg-purple-600 hover:bg-purple-700 text-white font-semibold rounded-md transition duration-300 shadow-md shadow-purple-900/50"
                        >
                            Log Contribution
                        </button>
                    </form>

                    {/* List of Contributions */}
                    <div className="space-y-3 max-h-96 overflow-y-auto">
                        {contributions.length === 0 ? (
                            <p className="text-gray-500 italic">No protocols logged yet.</p>
                        ) : (
                            contributions.map(c => (
                                <div key={c.id} className="p-3 bg-gray-700 rounded-lg border border-gray-600">
                                    <div className="flex justify-between items-start">
                                        <div>
                                            <h4 className="font-semibold text-white">{c.title}</h4>
                                            <p className="text-xs text-purple-300/80">Target: {c.pillar_target}</p>
                                        </div>
                                        <button
                                            onClick={() => handleDeleteContribution(c.id, c.title)}
                                            className="text-red-400 hover:text-red-500 text-sm px-2 py-1 rounded transition duration-200"
                                        >
                                            <svg xmlns="http://www.w3.org/2000/svg" className="h-4 w-4" fill="none" viewBox="0 0 24 24" stroke="currentColor" strokeWidth={2}>
                                                <path strokeLinecap="round" strokeLinejoin="round" d="M19 7l-.867 12.142A2 2 0 0116.138 21H7.862a2 2 0 01-1.995-1.858L5 7m5 4v6m4-6v6m1-10V4a1 1 0 00-1-1h-4a1 1 0 00-1 1v3M4 7h16" />
                                            </svg>
                                        </button>
                                    </div>
                                    <p className="text-xs text-gray-400 mt-2 overflow-hidden whitespace-nowrap text-ellipsis">
                                        {c.content}
                                    </p>
                                </div>
                            ))
                        )}
                    </div>
                </section>
            </main>
        </div>
    );
};

export default App;
