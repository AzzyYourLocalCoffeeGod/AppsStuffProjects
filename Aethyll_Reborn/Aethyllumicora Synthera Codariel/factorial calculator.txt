import React, { useState } from 'react';

// The main App component that contains the entire application logic and UI.
const App = () => {
  // useState hook to store the number entered by the user.
  const [number, setNumber] = useState('');
  // useState hook to store the calculated factorial result.
  const [result, setResult] = useState('');
  // useState hook to store any error messages for invalid input.
  const [error, setError] = useState('');

  // Function to calculate the factorial of the number.
  const calculateFactorial = () => {
    // Clear any previous error messages and results.
    setError('');
    setResult('');

    // Convert the input string to an integer.
    const num = parseInt(number, 10);

    // Validate the input.
    if (isNaN(num) || num < 0 || num > 100) {
      // Set an error message if the input is not a non-negative integer or is too large.
      setError('Please enter a non-negative integer less than or equal to 100.');
      return;
    }

    // Handle the base case for 0!
    if (num === 0) {
      setResult('The factorial of 0 is 1.');
      return;
    }

    // Initialize the factorial calculation with a BigInt to handle large numbers.
    let factorial = BigInt(1);
    // Loop from 1 up to the number to calculate the factorial.
    for (let i = 1; i <= num; i++) {
      factorial *= BigInt(i);
    }
    
    // Set the final result with a formatted string.
    setResult(`The factorial of ${num} is ${factorial.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ",")}.`);
  };

  return (
    <div className="flex flex-col items-center justify-center min-h-screen bg-gray-100 p-4 font-sans">
      <div className="bg-white p-8 rounded-2xl shadow-lg w-full max-w-md text-center">
        <h1 className="text-3xl font-bold mb-4 text-gray-800">Factorial Calculator</h1>
        <p className="text-gray-600 mb-6">
          Enter a non-negative integer to calculate its factorial.
        </p>

        {/* Input field for the number */}
        <input
          type="number"
          value={number}
          onChange={(e) => setNumber(e.target.value)}
          onKeyDown={(e) => {
            // Trigger calculation on 'Enter' key press.
            if (e.key === 'Enter') {
              calculateFactorial();
            }
          }}
          placeholder="Enter a number (0-100)"
          className="w-full p-3 mb-4 text-center text-gray-700 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500 transition-all duration-200"
        />

        {/* Button to trigger the calculation */}
        <button
          onClick={calculateFactorial}
          className="w-full bg-blue-600 text-white font-semibold py-3 px-6 rounded-lg hover:bg-blue-700 focus:outline-none focus:ring-2 focus:ring-blue-500 focus:ring-opacity-50 transition-all duration-200 shadow-md"
        >
          Calculate Factorial
        </button>

        {/* Conditional rendering for the error message */}
        {error && (
          <div className="mt-4 p-3 bg-red-100 text-red-700 border border-red-300 rounded-lg">
            <p className="font-medium">{error}</p>
          </div>
        )}

        {/* Conditional rendering for the result message */}
        {result && (
          <div className="mt-4 p-4 bg-green-100 text-green-700 border border-green-300 rounded-lg">
            <p className="font-semibold">{result}</p>
          </div>
        )}
      </div>
    </div>
  );
};

export default App;
