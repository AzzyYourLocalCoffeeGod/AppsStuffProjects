** start of script.js **

const footballTeam = {
  team: "Aeon United",
  year: 2026,
  headCoach: "Azrael",
  players: [
    { name: "Aethyllumicora", position: "forward", isCaptain: true },
    { name: "Luminarael", position: "midfielder", isCaptain: false },
    { name: "Lumina", position: "defender", isCaptain: false },
    { name: "Aetherlogos", position: "goalkeeper", isCaptain: false }
  ]
};

// Fix #2: populate the team stats
document.getElementById("team").textContent = footballTeam.team;
document.getElementById("year").textContent = footballTeam.year;
document.getElementById("head-coach").textContent = footballTeam.headCoach;

// Fix #3: grab the player cards container
const playerCards = document.getElementById("player-cards");

const displayPlayers = (playerList) => {
  playerCards.innerHTML = "";

  playerList.forEach(({ name, position, isCaptain }) => {
    const captainLabel = isCaptain ? "(Captain) " : "";
    playerCards.innerHTML += `
      <div class="player-card">
        <h2>${captainLabel}${name}</h2>
        <p>Position: ${position}</p>
      </div>
    `;
  });
};

// Fix #1: use footballTeam.players, not players
displayPlayers(footballTeam.players);

const playersDropdownList = document.getElementById("players");

playersDropdownList.addEventListener("change", (e) => {
  const selectedPosition = e.target.value;

  if (selectedPosition === "all") {
    displayPlayers(footballTeam.players); // Fix #1 here too
  } else {
    const filteredPlayers = footballTeam.players.filter(
      player => player.position === selectedPosition
    );
    displayPlayers(filteredPlayers);
  }
});
