function sumFibs(num) {
  let prevNumber = 0;
  let currNumber = 1;
  let result = 0;

  while (currNumber <= num) {
    if (currNumber % 2 !== 0) {
      result += currNumber;
    }

    
    let nextNumber = prevNumber + currNumber; 
    
    
    prevNumber = currNumber;
    currNumber = nextNumber;
  }
  
  return result;
}
