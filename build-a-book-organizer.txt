// 1. Array of book objects (minimum 3)
const books = [
  { title: "Dune", authorName: "Frank Herbert", releaseYear: 1965 },
  { title: "1984", authorName: "George Orwell", releaseYear: 1949 },
  { title: "Foundation", authorName: "Isaac Asimov", releaseYear: 1951 },
  { title: "Neuromancer", authorName: "William Gibson", releaseYear: 1984 },
];

// 2. Callback for sorting by year (ascending = smaller year first)
function sortByYear(bookA, bookB) {
  if (bookA.releaseYear < bookB.releaseYear) return -1;
  if (bookA.releaseYear > bookB.releaseYear) return 1;
  return 0;
}

// 3. Filter books after 1950
const filteredBooks = books.filter(book => book.releaseYear <= 1950);

// 4. Sort the filtered array (mutates it)
filteredBooks.sort(sortByYear);

// Optional: log to see result
console.log("Books after 1950, sorted by year:", filteredBooks);
