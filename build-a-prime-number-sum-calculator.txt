function sumPrimes(num) {
  // 1. Helper function to check if a number is prime
  function isPrime(n) {
    if (n < 2) return false;
    for (let i = 2; i <= Math.sqrt(n); i++) {
      if (n % i === 0) return false; // If it's divisible by anything else, it's out!
    }
    return true;
  }

  // 2. Loop through all numbers up to 'num' and sum the primes
  let sum = 0;
  for (let i = 2; i <= num; i++) {
    if (isPrime(i)) {
      sum += i;
    }
  }
  
  return sum;
}
