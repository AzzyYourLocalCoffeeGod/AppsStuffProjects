let inventory = []; // Your empty warehouse

function findProductIndex(name) {
  const searchName = name.toLowerCase();
  // We use a loop to check each "box" in the warehouse
  for (let i = 0; i < inventory.length; i++) {
    if (inventory[i].name.toLowerCase() === searchName) {
      return i; // Found it! Here is the shelf number.
    }
  }
  return -1; // Searched everywhere, not here.
}

function addProduct(product) {
  const index = findProductIndex(product.name);
  
  if (index !== -1) {
    // Product exists, just add the new quantity to the old one
    inventory[index].quantity += product.quantity;
    console.log(product.name.toLowerCase() + " quantity updated");
  } else {
    // New product! Make sure the name is saved as lowercase for consistency
    product.name = product.name.toLowerCase();
    inventory.push(product);
    console.log(product.name + " added to inventory");
  }
}


function removeProduct(name, quantity) {
  const index = findProductIndex(name);
  const lowerName = name.toLowerCase(); // Create a lowercase version for the logs

  if (index === -1) {
    console.log(lowerName + " not found"); // Fixes Test 12
    return;
  }

  const currentProduct = inventory[index];

  if (currentProduct.quantity < quantity) {
    // Fixes Test 16
    console.log("Not enough " + lowerName + " available, remaining pieces: " + currentProduct.quantity);
  } else {
    currentProduct.quantity -= quantity;
    
    let currentQty = currentProduct.quantity; // Save qty before potentially deleting

    if (currentProduct.quantity === 0) {
      inventory.splice(index, 1);
    }
    // Fixes Test 14
    console.log("Remaining " + lowerName + " pieces: " + currentQty);
  }
}
