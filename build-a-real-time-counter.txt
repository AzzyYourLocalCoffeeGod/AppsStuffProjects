** start of script.js **

/* file: script.js */
const textInput = document.getElementById("text-input");
const charCount = document.getElementById("char-count");

function updateCharacterCount() {
    // 1. Get the current text
    let text = textInput.value;
    
    // 2. Trim if over 50 (Requirement 6)
    if (text.length > 50) {
        textInput.value = text.substring(0, 50);
    }

    // 3. Update the count string exactly (Requirement 4)
    const updatedCount = textInput.value.length;
    charCount.textContent = "Character Count: " + updatedCount + "/50";

    // 4. The Style Property Fix (Requirement 5)
    if (updatedCount >= 50) {
        // We use the hex for pure red. If this fails, try "red"
        charCount.style.color = "#ff0000"; 
    } else {
        // Reset to default black/grey
        charCount.style.color = ""; 
    }
}

textInput.addEventListener("input", updateCharacterCount);

// Initialize for the 0/50 state
updateCharacterCount();

** end of script.js **
