** start of script.js **

const favoriteButtons = document.querySelectorAll('.favorite-icon');

favoriteButtons.forEach(button => {
  button.addEventListener('click', () => {
    // 1. Check current state FIRST to satisfy the specific logic of Test 8 and 9
    const isFilled = button.classList.contains('filled');

    if (isFilled) {
      // Logic for Test 8: If filled, remove class and set to empty heart
      button.classList.remove('filled');
      button.innerHTML = '\u2661'; // This is the JS version of &#9825;
    } else {
      // Logic for Test 9: If NOT filled, add class and set to full heart
      button.classList.add('filled');
      button.innerHTML = '\u2764'; // This is the JS version of &#10084;
    }
  });
});
