** start of script.js **

const switcherBtn = document.getElementById("theme-switcher-button");
const dropdown = document.getElementById("theme-dropdown");
const announcer = document.getElementById("theme-announcer");
const body = document.body;

// Story 17, 18, 19: The Themes Array
const themes = [
    { name: "neon", message: "AesiaLux Neon Core: Activated." },
    { name: "obsidian", message: "Deep Obsidian Silence: Engaged." }
];

// Logic for opening/closing the dropdown
switcherBtn.addEventListener("click", () => {
    const isExpanded = switcherBtn.getAttribute("aria-expanded") === "true";
    
    // Toggle state
    dropdown.hidden = isExpanded;
    switcherBtn.setAttribute("aria-expanded", !isExpanded);
});

// Logic for selecting a theme
const themeItems = document.querySelectorAll("#theme-dropdown li");

themeItems.forEach((item) => {
    item.addEventListener("click", () => {
        const themeName = item.innerText.toLowerCase();
        
        // Story 26: Remove old theme classes and add new one
        body.className = ""; // Reset
        body.classList.add(`theme-${themeName}`);
        
        // Story 27: Find message and update aria-live
        const themeData = themes.find(t => t.name === themeName);
        announcer.textContent = themeData.message;
        
        // Close menu after selection
        dropdown.hidden = true;
        switcherBtn.setAttribute("aria-expanded", "false");
    });
});

** end of script.js **
