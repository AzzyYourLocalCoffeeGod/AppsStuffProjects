// 1 & 2: array with at least five question objects
const questions = [
  {
    category: "Programming Languages",
    question: "Why did Azzy invent AEsiaLux?",
    choices: ["It's easy", "It's the future", "He loves machines"],  // ← only 3 different ones
    answer: "He loves machines"
  },
  {
    category: "JavaScript",
    question: "What does console.log() do?",
    choices: ["Prints to the screen", "Sends data to the server", "Creates a variable"],
    answer: "Prints to the screen"
  },
  {
    category: "Arrays",
    question: "Which method adds to the end of an array?",
    choices: ["push()", "unshift()", "pop()"],
    answer: "push()"
  },
  {
    category: "Objects",
    question: "How do you access a property called 'name'?",
    choices: ["obj[name]", "obj.name", "obj->name"],
    answer: "obj.name"
  },
  {
    category: "Functions",
    question: "What does a function return by default if no return is written?",
    choices: ["null", "undefined", "0"],
    answer: "undefined"
  }
];;

// 7. Returns a random question object from the array
function getRandomQuestion(questionsArray) {
  if (questionsArray.length === 0) return null;
  const randomIndex = Math.floor(Math.random() * questionsArray.length);
  return questionsArray[randomIndex];
}

// 8. Returns a random choice from the choices array
function getRandomComputerChoice(choicesArray) {
  if (choicesArray.length === 0) return null;
  const randomIndex = Math.floor(Math.random() * choicesArray.length);
  return choicesArray[randomIndex];
}

// 9. Returns result string based on computer's choice
function getResults(questionObj, computerChoice) {
  if (computerChoice === questionObj.answer) {
    return "The computer's choice is correct!";
  } else {
    return `The computer's choice is wrong. The correct answer is: ${questionObj.answer}`;
  }
}

// Optional test calls (delete if lab doesn't want them)
console.log(getRandomQuestion(questions));
console.log(getRandomComputerChoice(["a", "b", "c"]));
console.log(getResults(questions[0], "He loves machines")); // correct
console.log(getResults(questions[0], "He was bored")); // wrong
