// 1. Declare num and assign a number between 1 and 20
const num = 5;  // You can pick any number 1–20, e.g. 7, 10, 12, etc.

// 2 & 3. Create the function factorialCalculator
function factorialCalculator(number) {
  let result = 1;
  
  // Loop from 1 to number (inclusive)
  for (let i = 1; i <= number; i++) {
    result = result * i;  // or result *= i;
  }
  
  return result;
}

// 4. Call the function with num and store the result
const factorial = factorialCalculator(num);

// 5. Create the output message in the exact format they want
const resultMsg = `Factorial of ${num} is ${factorial}`;

// 6. Output to console
console.log(resultMsg);
