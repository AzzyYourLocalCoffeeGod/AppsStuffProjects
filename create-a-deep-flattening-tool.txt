function steamrollArray(arr) {
  const flattenedArray = [];

  // Create a helper function or just use a loop
  for (let i = 0; i < arr.length; i++) {
    if (Array.isArray(arr[i])) {
      // It's a box! Dig deeper by calling steamrollArray again
      // We use the spread operator (...) to pull the items out of the result
      flattenedArray.push(...steamrollArray(arr[i]));
    } else {
      // It's a loose item (even an empty object {}). Just grab it.
      flattenedArray.push(arr[i]);
    }
  }

  return flattenedArray;
}
