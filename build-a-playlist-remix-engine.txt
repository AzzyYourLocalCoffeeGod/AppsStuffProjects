function flattenPlaylists(playlists) {
if (!Array.isArray(playlists)) {
  return [];
}
const result = [];
for (let i = 0; i < playlists.length; i++) {
  for (let j = 0; j < playlists[i].length; j++) {
    const track = playlists[i][j];
    result.push({
      ...track, source: [i, j]
    });
  }
}
return result;
};
function scoreTracks(tracks) {
const result = [];
for (let track of tracks) {
  result.push({
    ...track, 
    score: track.votes * 10 - Math.abs(track.bpm - 120)
  });
}
return result;
};
function dedupeTracks(tracks) {
  const seen = new Set();
  const result = [];
  for (let track of tracks) {
    if (!seen.has(track.trackId)) {
      seen.add(track.trackId);
      result.push(track);
    }
  }
  return result;
};
function enforceArtistQuota(tracks, maxPerArtist) {
const artistCount = {};
const result = [];
for (let track of tracks) {
if (artistCount[track.artist] === undefined) {
  artistCount[track.artist] = 0;
}
if (artistCount[track.artist] < maxPerArtist) {
  result.push(track);
  artistCount[track.artist]++;
}
}
return result;
};
function buildSchedule(tracks) {
const result = [];
for (let i = 0; i < tracks.length; i++) {
  result.push({
    slot: i + 1,
    trackId: tracks[i].trackId
  });
}
return result;
};
function remixPlaylist(playlists, maxPerArtist) {
  const step1 = flattenPlaylists(playlists);
  const step2 = scoreTracks(step1);
  const step3 = dedupeTracks(step2);
  const step4 = enforceArtistQuota(step3, maxPerArtist);
  const step5 = buildSchedule(step4);
  return step5;
}
