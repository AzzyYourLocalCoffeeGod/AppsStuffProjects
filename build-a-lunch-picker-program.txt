const lunches = [];
function addLunchToEnd(lunches, item) {
  lunches.push(item);                              // ← use the parameter!
  console.log(`${item} added to the end of the lunch menu.`);  // ← lowercase item
  return lunches;
}
// 6–9: addLunchToStart (almost identical, just unshift instead of push)
function addLunchToStart(lunches, item) {
  lunches.unshift(item);
  console.log(`${item} added to the start of the lunch menu.`);
  return lunches;
}

// 10–14: removeLastLunch (use pop, check for empty)
function removeLastLunch(lunches) {
  if (lunches.length === 0) {
    console.log("No lunches to remove.");
  } else {
    let removed = lunches.pop();
    console.log(`${removed} removed from the end of the lunch menu.`);
  }
  return lunches;
}

// 15–19: removeFirstLunch (use shift, check for empty)
function removeFirstLunch(lunches) {
  if (lunches.length === 0) {
    console.log("No lunches to remove.");
  } else {
    let removed = lunches.shift();
    console.log(`${removed} removed from the start of the lunch menu.`);
  }
  return lunches;
}

// 21–25: getRandomLunch (pick random, log it, DO NOT change array)
function getRandomLunch(lunches) {
  if (lunches.length === 0) {
    console.log("No lunches available.");
  } else {
    let randomIndex = Math.floor(Math.random() * lunches.length);
    let item = lunches[randomIndex];
    console.log(`Randomly selected lunch: ${item}`);
  }
  // Important: do NOT modify lunches — just read from it
}

// 26–30: showLunchMenu (join with ", " or empty message)
function showLunchMenu(lunches) {
  if (lunches.length === 0) {
    console.log("The menu is empty.");
  } else {
    console.log(`Menu items: ${lunches.join(", ")}`);
  }
}
