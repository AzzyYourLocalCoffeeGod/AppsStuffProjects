** start of script.js **

const textInput = document.getElementById("text-input");
const checkBtn = document.getElementById("check-btn");
const result = document.getElementById("result");

checkBtn.addEventListener("click", () => {
  const original = textInput.value;

  if (original === "") {
    alert("Please input a value");
    return;
  }

  // Remove non-alphanumeric characters and lowercase everything
  const cleaned = original.replace(/[^a-z0-9]/gi, "").toLowerCase();
  const reversed = cleaned.split("").reverse().join("");

  if (cleaned === reversed) {
    result.textContent = `${original} is a palindrome`;
  } else {
    result.textContent = `${original} is not a palindrome`;
  }
});
