// The taxRate is "optional" because we set it to 0.07 by default.
function calculateTotalBill(mealPrice, taxRate = 0.07) {
  // Logic: Price + (Price * Tax)
  const total = mealPrice + (mealPrice * taxRate);
  return total;
}

// Determine the meal's total (e.g., a meal costing 100 credits)
const mealCost = 100;
const finalBill = calculateTotalBill(mealCost);

console.log(`The final bill for your cosmic feast is ${finalBill} credits.`);
