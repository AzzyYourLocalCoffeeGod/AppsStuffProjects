function updateRecords(records, id, prop, value) {
  // Case 1: delete if value is empty string
  if (value === "") {
    delete records[id][prop];
  }
  // Case 2: if prop is NOT "tracks", just set the value
  else if (prop !== "tracks") {
    records[id][prop] = value;
  }
  // Case 3 & 4: prop is "tracks" and value is not empty
  else {
    // If no tracks array yet, create it
    if (!records[id].tracks) {
      records[id].tracks = [];
    }
    // Always push the value to tracks
    records[id].tracks.push(value);
  }

  // Always return the entire records object
  return records;
}
