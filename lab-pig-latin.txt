function translatePigLatin(str) {
  const vowelRegex = /[aeiou]/;
  if (vowelRegex.test(str[0])) {
    return str + "way";
  }
  const firstVowelIndex = str.indexOf(str.match(vowelRegex));
  if (firstVowelIndex === -1) {
    return str + "ay";
  }
  const start = str.substring(firstVowelIndex);
  const end = str.substring(0, firstVowelIndex);
  return start + end + "ay"
}
